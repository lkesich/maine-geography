 52## Townships
 53# Constants
 54REGIONS: List[str] = ['ED','MD','ND','SD','TS','BKP','BPP','EKR','NWP','WKR','NBKP','NBPP','WBKP','WELS']
 55""" Valid region codes that can appear in township names.
 56    
 57These are two- to four-letter codes like 'WELS' (West of the
 58Easterly Line of the State) or 'BPP' (Bingham's Penobscot Purchase).
 59
 60Township names can contain zero, one, or two regions."""
 61
 62# Building blocks
 63REGION: str = f"(?:(?<![a-z])(?:{'|'.join(REGIONS)})(?![a-z]))"
 64""" Uncompiled regex building block representing a region designator."""
 65
 66RANGE: str = f"(?:R.?[\\d]{{1,2}})"
 67""" Uncompiled regex building block representing a range designator.
 68
 69Ranges are counted from the easterly line toward the west,
 70numbered 1-19 (e.g., R1, R19)."""
 71
 72TOWNSHIP_STANDARD: str = "(?:T.?\\d{1,2})"
 73TOWNSHIP_ALTERNATE: str = f"(?<!\\w)T[ABCDX](?![a-z])"
 74TOWNSHIP: str = f"(?:{TOWNSHIP_STANDARD}|{TOWNSHIP_ALTERNATE})"
 75""" Uncompiled regex building block representing a township designator.
 76
 77Townships are designated by:
 78    - Numbers 1-19 from south to north (e.g., T1, T19)
 79    - Occasional letter designations (TA, TB, TC, TD, TX)"""
 80
 81UNNAMED: str = f"((?:{TOWNSHIP})(?:{FUZZY}{RANGE})?(?:{FUZZY}{REGION}){{0,2}})"
 82UNNAMED_ELEMENTS: str = f"(?:{'|'.join([TOWNSHIP, RANGE, REGION])})"
 83
 84# Patterns
 85UNNAMED_PATTERN: re.Pattern = re.compile(UNNAMED, re.I)
 86""" Compiled regex matching a full unnamed township name, with tolerance for formatting variation.
 87
 88Used in `mainegeo.townships.is_unnamed_township` and `mainegeo.townships.clean_codes`."""
 89
 90UNNAMED_ELEMENTS_PATTERN: re.Pattern = re.compile(UNNAMED_ELEMENTS, re.I)
 91""" Compiled regex matching any unnamed township name element.
 92
 93Used in `mainegeo.townships.clean_code`."""
 94
 95LAST_REGION_PATTERN: re.Pattern = re.compile(f" {REGION}$", re.I)
 96""" Compiled regex matching the last region code in an unnamed township name.
 97
 98Used in `mainegeo.townships.strip_region`."""

100## Result parsing
101# Building blocks
102LEADING_ZERO: str = '(?<=[^\\d])0(?=\\d)'
103NOT_REPORTING: str = f'(?!AND|&)'
104PARENTHETICAL: str = f'\\({NOT_REPORTING}[^\\(]+\\)'
105PRECEDES_DASH: str = f'^[^-]+--'
106
107STANDARD_DELIMITER: str = ", "
108""" Preferred delimiter character; all other delimiters will be replaced by this."""
109
110NONSTANDARD_DELIMITERS: str = ["&", "/", "(AND ", "(INC "]
111""" Substrings that are sometimes used by the SoS to delimit reporting towns."""
112
113MEANINGFUL_CHARACTERS: List[str] = ["(", ")", "-", "&", "/", ","]
114""" Punctuation characters used by the SoS to communicate info about a reporting unit.
115
116Examples:
117    * BENEDICTA/SILVER RIDGE TWPS: forward slashes delimit towns.
118    * T15 R6 TWP (EAGLE LAKE): parentheses indicate registration town
119    * BLAINE -- E TWP: double hyphens indicate registration town
120
121These characters should not be stripped until this info has been parsed out."""
122
123# Patterns
124REGISTRATION_PATTERN: re.Pattern = re.compile(f'{PARENTHETICAL}|{PRECEDES_DASH}', re.I)
125""" Compiled regex matching a substring of non-reporting registration towns.
126
127Used in `mainegeo.elections.ResultString`.
128"""
129
130CLEAN_TOWNSHIP_PATTERN: re.Pattern = re.compile(f"[^\\w]|{LEADING_ZERO}")
131""" Compiled regex matching non-word characters and leading zeroes.
132
133Used in `mainegeo.townships.clean_code`."""
134
135NON_ALIAS_PATTERN: re.Pattern = re.compile(
136    f'{UNNAMED}(?: twps?)?|{PUNCTUATION}',
137    re.I
138    )
139""" Compiled regex matching substrings that are not an unnamed township code or punctuation.
140
141Used in `mainegeo.townships.has_alias` and `mainegeo.townships.extract_alias`."""
142
143DROP_CHARACTERS_PATTERN: re.Pattern = re.compile(
144    f"[^\\w\\s{''.join(map(re.escape, MEANINGFUL_CHARACTERS))}]"
145    )
146""" Matches all characters except word characters, whitespace, and meaningful punctuation.
147
148Meaningful punctuation characters are those used by the SoS to communicate information
149about a reporting unit (e.g. forward slashes to separate reporting towns). The full list
150of meanginful characters is defined in `mainegeo.patterns.MEANINGFUL_CHARACTERS`.
151
152Used in `mainegeo.elections.ResultString._drop_non_meaningful_characters`."""
153
154NONSTANDARD_DELIMITER_PATTERN: re.Pattern = re.compile(
155    '|'.join(map(re.escape, NONSTANDARD_DELIMITERS)),
156    re.I
157    )
158""" Compiled regex matching non-standard result string delimiters used occasionally by the SoS.
159
160Used in `mainegeo.elections.ResultString`."""
161
162ORPHAN_PARENTHESIS_PATTERN: re.Pattern = re.compile(
163    f'^(?P<result>[^(]+)(?P<orphan_parenthesis>[)])$'
164    )
165""" Compiled regex matching the orphaned closing parenthesis left after delimiter normalization.
166
167Capture groups:
168    * result
169    * orphan_parenthesis
170
171Used in `mainegeo.elections.ResultString`."""

173## Name standardization
174# Constants
175GNIS_GEOTYPES: List[str] = ["CITY", "PLANTATION", "TOWNSHIP", "TOWN"]
176""" Geotypes used by the Geographic Names Information System (GNIS)."""
177
178ABBREVIATIONS: Dict[str, str] = {
179    "PLANTATION":       "PLT",
180    "TOWNSHIP":         "TWP",
181    "VOTING DISTRICT":  "VOTING DIST",
182    "RESERVATION":      "RES"
183}
184""" Geotype suffixes used by the Maine SoS and their abbreviations."""
185
186JUNIOR_SUFFIXES: List[str] = ['GORE', 'GRANT', 'ISLAND']
187""" Geotypes which may precede another geotype suffix or be used alone.
188
189For example, Moxie Gore and Moxie Gore Twp have subtly different meanings,
190but refer to the same place and are often used interchangeably by the SoS."""
191
192DIRECTIONS: List[str] = ['NORTH', 'SOUTH', 'EAST', 'WEST']
193""" Direction words which may modify place names."""
194
195CONTAINS_FALSE_SUFFIX: List[str] = ['INDIAN TOWNSHIP', 'INDIAN RESERVATION']
196""" Canonical place names that contain a word that is normally a suffix.
197
198For example: Indian Township is the name of a town, not a township.
199These false suffixes should be treated differently than true 
200suffixes during processing."""
201
202AMBIGUOUS_SUFFIXES: List[str] = ['RES']
203""" Full or abbreviated suffixes that may occur as substrings in other contexts. """
204
205# Factory functions
206def generate_suffixes() -> dict[str, str]:
207    replacements = []
208    for name, abbr in ABBREVIATIONS.items():
209        
210        precedes_false = []
211        for town in CONTAINS_FALSE_SUFFIX:
212            if town.endswith((name, abbr)):
213                precedes_false.append(re.sub(f' {name}|{abbr}$', '', town)) 
214        
215        if len(precedes_false) > 0:
216            ignore_false = f"(?<!{'|'.join(precedes_false)} )"
217        else:
218            ignore_false = ''
219        
220        for suffix in (name, abbr):
221            required_whitespace = ' ' if suffix in AMBIGUOUS_SUFFIXES else ' ?'
222            terminator = '$' if suffix in AMBIGUOUS_SUFFIXES else 's?$'
223            
224            replacements.append({
225                'suffix': suffix,
226                'replacement': ' ' + abbr,
227                'strip_pattern': required_whitespace + ignore_false + suffix + terminator,
228                'clean_pattern': required_whitespace + suffix + f"(?={terminator})",
229                'terminator': terminator
230            })
231    
232    return replacements
233
234def generate_valid_punctuation(char: str, template: str) -> str:
235    pattern = re.compile(f'(?P<leading>\\w+ ?){char}(?P<trailing> ?\\w+)')
236    matches = map(pattern.match, TOWNSHIPS.town)
237    valid_contexts = [match.expand(template) for match in matches if match]
238    return '|'.join(valid_contexts)
239
240# Templates
241AMPERSANDS_TEMPLATE: str = '(?:(?<=\g<leading>)&(?=\g<trailing>))'
242HYPHENS_TEMPLATE: str = '\g<leading>-(?=\g<trailing>)'
243
244# Building blocks
245GNIS_NAME = f"(?P<geotype>{'|'.join(GNIS_GEOTYPES)}) of (?P<town>.+)"
246VALID_AMPERSANDS: str = generate_valid_punctuation('&', AMPERSANDS_TEMPLATE)
247VALID_HYPHENS: str = generate_valid_punctuation('-', HYPHENS_TEMPLATE)
248JUNIOR_SUFFIX: str = f"\\b({'|'.join(JUNIOR_SUFFIXES)})"
249SUFFIXES: list[dict] = generate_suffixes()
250SUFFIX_REPLACEMENTS: dict[str, str] = {
251    suffix['clean_pattern']: suffix['replacement']
252    for suffix in SUFFIXES
253}
254
255# Patterns
256GNIS_PATTERN: re.Pattern = re.compile(GNIS_NAME, re.I)
257""" Matches place names with Geographic Names Information System (GNIS) formatting.
258
259Capture groups:
260    * geotype
261    * town
262
263Used in `mainegeo.townships.normalize_suffix`."""
264
265SUFFIX_PATTERN: re.Pattern = re.compile(
266    f"{'|'.join([suffix['strip_pattern'] for suffix in SUFFIXES])}", re.I
267    )
268""" Matches all valid terminal suffixes and suffix abbreviations.
269
270Used in `mainegeo.townships.strip_suffix`."""
271
272VALID_AMPERSANDS_PATTERN: re.Pattern = re.compile(VALID_AMPERSANDS, re.I)
273""" Matches ampersands that are part of canonical town names and should
274not be interpreted as SoS result string formatting.
275
276Used in `mainegeo.townships.strip_town` (a helper for `mainegeo.townships.clean_town`)."""
277
278INVALID_PUNCTUATION_PATTERN: re.Pattern = re.compile(
279    f"{PUNCTUATION}(?<!{VALID_HYPHENS})", re.I
280    )
281""" Matches all punctuation except hyphens that are part of canonical town names.
282
283Used in `mainegeo.townships.strip_town` (a helper for `mainegeo.townships.clean_town`)."""
284
285ENDSWITH_JUNIOR_SUFFIX_PATTERN: re.Pattern = re.compile(f".+{JUNIOR_SUFFIX}$", re.I)

289
290Used in `mainegeo.townships.toggle_suffix`."""
291
292CONTAINS_JUNIOR_SUFFIX_PATTERN: re.Pattern = re.compile(f".+{JUNIOR_SUFFIX} TWP$", re.I)
293""" Matches town name that contains a junior suffix (gore, grant, island, etc).

295Junior suffixes are listed in `mainegeo.patterns.JUNIOR_SUFFIXES`.
296
297Used in `mainegeo.townships.toggle_suffix`."""

300## Unspecified groups
301# Constants
302UNSPECIFIED_FLAG: str = 'TWPS'
303""" Substring that indicates an unspecified group may be present in a raw election result.
304
305Used in multiple functions in `mainegeo.elections`. Most important use is in
306`mainegeo.elections.ReportingUnit.has_unspecified_group`, which uses it in 
307combination with other context clues to detect unspecified groups."""
308
309STANDARD_FLAG: str = 'UNSPECIFIED'
310""" Substring that will be applied to unspecified groups during formatting.
311
312Chosen to avoid overlap with words that may occur naturally in election 
313result strings."""
314
315MULTI_COUNTY_REGISTRATION_TOWNS: List[str] = ['MILLINOCKET']
316""" Towns that host unspecified township groups from multiple counties.
317
318As of 2025, the only town that is typically reported this way is Millinocket,
319e.g. Millinocket Penobscot Twps and Millinocket Piscataquis Twps."""
320
321# Building blocks
322PLURAL = UNSPECIFIED_FLAG
323SINGULAR = re.sub('S$', '', UNSPECIFIED_FLAG)
324UNSPECIFIED_REGTOWN = f"(?P<regtown>{'|'.join(MULTI_COUNTY_REGISTRATION_TOWNS)})"
325UNSPECIFIED_COUNTY = f"(?P<cty>{'|'.join(COUNTIES.sos_county)})"
326SOS_FLAG = f"(?P<sos_flag>{UNSPECIFIED_FLAG})"
327
328# Patterns
329PLURAL_PATTERN: re.Pattern = re.compile(f'\\b{PLURAL}\\b')
330SINGULAR_PATTERN: re.Pattern = re.compile(f'\\b{SINGULAR}\\b')
331MULTI_COUNTY_PATTERN: re.Pattern = re.compile(
332    f"(?i){UNSPECIFIED_REGTOWN} {UNSPECIFIED_COUNTY}\\w* {SOS_FLAG}", re.I
333)
334""" Matches raw, unformatted unspecified groups that contain a county.
335
336Capture groups:
337    * regtown
338    * cty
339    * sos_flag
340"""
341
342MULTI_COUNTY_FORMAT: str = r"\g<regtown> \g<sos_flag> [\g<cty>]"
343""" Standardized format to apply to all multi-county unspecified groups."""
344
345FORMATTED_GROUP_PATTERN: re.Pattern = re.compile(
346    f'{STANDARD_FLAG} (?P<regtown>.+) {UNSPECIFIED_FLAG}( \\[{UNSPECIFIED_COUNTY}\\])?',
347    re.I
348)
349""" Matches all formatted unspecified groups, including multi-county groups."""
